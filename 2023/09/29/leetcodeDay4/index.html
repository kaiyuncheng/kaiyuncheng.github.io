leetcode #561. Array Partition I

小排到大後，一對中的較小值 (因為已經排序好了所以就是一對中的第一個) 的 加總

var arrayPairSum = function (nums) {
  let len = nums.length;
  let sum = 0;
  nums = nums.sort((a, b) => a - b);
  for (let i = 0; i < len; i += 2) {
    sum += nums[i];
  }
  return sum;
};

arrayPairSum([6, 2, 6, 5, 1, 2, 7, 9]); // 1+2+6+7 = 16
